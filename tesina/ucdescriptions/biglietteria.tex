{\ucTable	% UC: Lista Prenotazioni
{	% Caso d'uso
	Lista Prenotazioni
	
}{	% ID
	\increaseCount{classId}
	
}{	% Breve descrizione
	L'utente visualizza l'elenco delle prenotazione salvate nel Sistema.
	
}{	% Attori Primari
	\attori
	
}{	% Precondizioni
	\begin{enumerate}
        \item L'utente è autenticato come Amministratore o Biglietteria.
    \end{enumerate}
	
}{	% Sequenza Eventi Principale
	\begin{enumerate}
        \item Il caso d'uso inizia quando l'utente richiede di visualizzare la lista completa delle prenotazione.
        \item \textbf{If} c'è almeno una prenotazione memorizzata nel Sistema 
        \begin{enumerate}
            \item \textbf{For} ogni prenotazione nel Sistema
            \begin{enumerate}
                \item Il Sistema mostra un'anteprima della prenotazione con informazioni basiche.
            \end{enumerate}
            \item[] \textbf{punto di estensione:} Ricerca Prenotazione
        \end{enumerate}
        \item \textbf{Else}
        \begin{enumerate}
            \item Il Sistema informa l'utente che non vi sono prenotazioni memorizzate.
        \end{enumerate}
    \end{enumerate}
	
}{	% Postcondizioni
	\begin{enumerate}
        \item L'utente visualizza l'elenco delle prenotazioni salvate nel Sistema.
    \end{enumerate}
	
}{	% Sequenza Eventi Alternativa
} }

{\ucExtTable	% UC EXT: Ricerca Prenotazione
{	% Caso d'uso
	Ricerca Prenotazione
	
}{	% ID
	\increaseCount{classId}
	
}{	% Breve descrizione
	L'utente filtra la lista delle prenotazioni applicando eventuali filtri.
	
}{	% Attori Primari
	\attori
	
}{	% Precondizioni
	\begin{enumerate}
        \item L'utente ha inserito i criteri di ricerca.
        \item L'utente ha premuto il pulsante di ricerca.
    \end{enumerate}
	
}{	% Sequenza Eventi Principale
	\begin{enumerate}
        \item Il Sistema ricerca, nell'elenco delle prenotazioni, delle prenotazioni coincidenti.
        \item \textbf{If} il Sistema trova almeno una prenotazione coincidente
        \begin{enumerate}
            \item \textbf{For} ogni prenotazione coincidente
            \begin{enumerate}
                \item Il Sistema mostra un'anteprima della prenotazione con informazioni basiche.
            \end{enumerate}
        \end{enumerate}
        \item \textbf{Else}
        \begin{enumerate}
            \item Il Sistema informa che non vi sono prenotazioni coincidenti.
        \end{enumerate}
    \end{enumerate}
	
}{	% Postcondizioni
	\begin{enumerate}
        \item L'utente visualizza l'elenco filtrato.
    \end{enumerate}
	
} }
