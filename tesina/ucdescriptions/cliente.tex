%--- CONVENZIONI NELLA GRAMMATICA DELLA TESINA: ---
% #1: Per pratticità, noi useremo include() e blocchi if nelle descrizioni dei casi d'uso.
%   - "include()" viene usato per indicare altri casi d'uso come passi nella sequenza degli eventi
%        principali di un casi d'uso in particolare.
%   - I "blocchi if" vengono usati per i passi dove si verifica il tipo di account dell'Utente.
%
% #2: Nella tabella del glossario ci devono essere 4 sezioni: Termine, Sinonimi, Definizione, Esempi d'uso.
%     - "Termine" contiene il termino canonico da utilizzare nelle descrizioni dei casi d'uso scritto in 
%         all-caps "PAROLA" (oppure come "Parola" se necessario). Se "parola" è presente nelle descrizioni,
%         si riferisce ad un'altro termine, e.g. sezione (dell'app) e Sezione (del teatro). 
%     - "Sinonimi" contiene tutte le possibili varianti trovate nelle descrizioni. Qui ci possono essere le
%         versioni come "Parola".
%     - "Definizione" contiene il significato inequivocabile del termine.
%     - "Esempi d'uso" contiene i casi un cui si può usare e trovare il termine, se necessario chiarirlo.
%
% #3: Le precondizioni possono usare questi tre tempi verbali:
%  	- Presente semplice per condizioni che devono verificarsi una volta, e.g. "si trova", "è autenticato".
% 	- Presente progresivo per condizioni che devono verificarsi continuamente, e.g. "sta visualizzando".
%
% #4: Sia le precondizioni che le postcondizioni devono seguire la struttura della Soggetto-Verbo-Oggetto.
%--- END

%--- CONVEZIONE NEI TERMINI USATI NELLA TESINA: ---
% La parola "sistema" viene scritta in minuscolo.
% Evitare l'uso di "dettagli" negli usecase di Visualizza ...(9).

% "memorizzate"(1) :  Deve avere un uso consistente in tutte le tabelle. Usato per indicare istanze
%   memorizzate nel model. Quando si parla di liste associate a un'istanza, e.g. le Regie associate all'Opera,
%    non è necessario usare questo termine.
% "INFORMAZIONI"(2) : In maiuscole, è il nome della sezione Info ed ha bisogno di una
%   definizione precisa.
% "mostra a schermo <oggetto>"(3) : Concetto usato nei passi realizzati dal sistema.
% "visualizza a schermo <oggetto>"(4) : Concetto usato nei passi realizzati dal'Utente.
% "si trova"(5) : Deve avere un uso consistente in tutte le tabelle.
% "associate"(6) : Il suo modo di uso è "lista [instanze A] associate a [instanza B]".
% "lista di <oggetto>"(7) : Deve usare sempre "di". No "delle" o "degli".
% "SPETTACOLI"(8) : In maiuscole, è il nome della sezione SPETTACOLI ed ha bisogno di
%   una definizione precisa.
% "visualizza in più dettaglio"(9) : Per indicare i casi d'uso Visualizza. Devo darli una
%   definizione megliore.

% "nel suo nome" : Usato in Ricerca Opera
% "nel suo titolo" : Usato in Ricerca Spettacolo
% "nel suo nominativo" : Usato in Ricerca Prenotazione

% Utilizzare "informazioni essenziali" per i casi d'uso di Lista.
% Utilizzare "informazioni" per i casi d'uso di Visualizza.

% "non vengono inclusi" : Usato per indicare le instanze che non sono visualizzate nei casi d'uso di ricerca.
% 'comando "Nome del comando"' : Qui, "Nome del comando" è un nome descrittivo del comando chiamato, e.g. "Crea Opera", "Genera ricevuta".
% "visualizza" : Usato negli usecase di Lista e Visualizza.
% "dati" : Usato negli usecase di Crea e Modifica quando si chiede di inserirli.
%---END

{\ucTable	% UC: Lista Opere
	{	% Caso d'uso
		Lista Opere
		
	}{	% ID
		\increaseCount{classId}
		
	}{	% Breve descrizione
		L'Utente visualizza la lista di Opere memorizzate.
        % "visualizza"
        % "memorizzate"(1)
		
	}{	% Attori Primari
		\attori
		
	}{	% Precondizioni
		\begin{enumerate}
			\item L'Utente si trova nella sezione INFORMAZIONI.
			% "INFORMAZIONI"(2)
			% "si trova"(5)
		\end{enumerate}
		
	}{	% Sequenza Eventi Principale
		\begin{enumerate}
			\item Il sistema mostra a schermo le informazioni essenziali di tutte le Opere memorizzate.
            % "mostra a schermo <oggetto>"(3)
            \item Il sistema permette all'Utente di visualizzare in più dettaglio le Opere.
            % "visualizza in più dettaglio"(9)
            \item \textbf{If} l'Utente è autenticato come Amministratore
            \begin{enumerate}
                \item Il sistema permette all'Utente di modificare ed eliminare le Opere.
            \end{enumerate}
            \item[] \textbf{punto di estensione:} Ricerca Opera
		\end{enumerate}
		
	}{	% Postcondizioni
		\begin{enumerate}
			\item La lista di Opere memorizzate viene visualizzata a schermo.
			% "visualizza a schermo <oggetto>"(4)
		\end{enumerate}
		
	}{	% Sequenza Eventi Alternativa
} }

	{	% Caso d'uso
		Crea Prenotazione
		
	}{	% ID
		\increaseCount{classId}
		
	}{	% Breve descrizione
		L'Utente prenota Posti di un'Evento svolto nel teatro.
		
	}{	% Attori Primari
		\attori
	}{	% Precondizioni
		\begin{enumerate}
			\item L'Utente si trova nella sezione Acquisto. % - È nessario aggiornari le descrizioni sugli Spettacoli
		\end{enumerate}
		
	}{	% Sequenza Eventi Principale
		\begin{enumerate}
			\item L'Utente avvia il caso d'uso selezionando il comando "Scegli posti".
			\item Il sistema, tramite un'interfaccia, permette di inserire i dati relativi alla Prenotazione, e.g. indicare un nominativo, selezionare un'Evento, ecc.
			\item L'Utente inserisce i dati relativi alla Prenotazione.
			\item L'Utente conferma la Prenotazione.
			\item Il sistema verifica i dati e li salva in memoria.
			% - Dovrebbe menzionare il caso StampaRicevuta?
		\end{enumerate}
		
	}{	% Postcondizioni
		\begin{enumerate}
			\item L'Utente crea una Prenotazione.
		\end{enumerate}
		
	}{	% Sequenza Eventi Alternativa
} }
